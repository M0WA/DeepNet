02332 /* Allocates the stack if it does not exist.
02333  *  Guarantees space for at least one push.
02334  */
02335 static void contentIndexer_yyensure_buffer_stack (yyscan_t yyscanner)
02336 {
02337         int num_to_alloc;
02338     struct yyguts_t * yyg = (struct yyguts_t*)yyscanner;
02339 
02340         if (!yyg->yy_buffer_stack) {
02341 
02342                 /* First allocation is just for 2 elements, since we don't know if this
02343                  * scanner will even need a stack. We use 2 instead of 1 to avoid an
02344                  * immediate realloc on the next call.
02345          */
02346                 num_to_alloc = 1;
02347                 yyg->yy_buffer_stack = (struct yy_buffer_state**)contentIndexer_yyalloc
02348                                                                 (num_to_alloc * sizeof(struct yy_buffer_state*)
02349                                                                 , yyscanner);
02350                 if ( ! yyg->yy_buffer_stack )
02351                         YY_FATAL_ERROR( "out of dynamic memory in contentIndexer_yyensure_buffer_stack()" );
02352                                                                   
02353                 memset(yyg->yy_buffer_stack, 0, num_to_alloc * sizeof(struct yy_buffer_state*));
02354                                 
02355                 yyg->yy_buffer_stack_max = num_to_alloc;
02356                 yyg->yy_buffer_stack_top = 0;
02357                 return;
02358         }
02359 
02360         if (yyg->yy_buffer_stack_top >= (yyg->yy_buffer_stack_max) - 1){
02361 
02362                 /* Increase the buffer to prepare for a possible push. */
02363                 int grow_size = 8 /* arbitrary grow size */;
02364

02366                 yyg->yy_buffer_stack = (struct yy_buffer_state**)contentIndexer_yyrealloc
02367                                                                 (yyg->yy_buffer_stack,
02368                                                                 num_to_alloc * sizeof(struct yy_buffer_state*)
02369                                                                 , yyscanner);
02370                 if ( ! yyg->yy_buffer_stack )
02371                         YY_FATAL_ERROR( "out of dynamic memory in contentIndexer_yyensure_buffer_stack()" );
02372 
02373                 /* zero only the new slots.*/
02374                 memset(yyg->yy_buffer_stack + yyg->yy_buffer_stack_max, 0, grow_size * sizeof(struct yy_buffer_state*));
02375                 yyg->yy_buffer_stack_max = num_to_alloc;
02376         }
